var  services = {
    "стрижка": "60 грн",
    "гоління": "80 грн",
    "Миття голови": "100 грн",};
 var methods={
price() {
    let sum = 0;
    for (let key in services) {
        if (typeof services[key] === "string" && services[key].includes("грн")) {
            sum += parseInt(services[key]); // Додаємо ціну до загальної суми
        }
    }
    return sum + " грн";
},
minPrice() {
    let prices = [];
    for (let key in services) {
        if (typeof services[key] === "string" && services[key].includes("грн")) {
            prices.push(parseInt(services[key])); // Зберігаємо ціни в масив
        }
    }
    return Math.min(...prices) + " грн"; // Знаходимо мінімальну ціну
},
maxPrice() {
    let prices = [];
    for (let key in services) {
        if (typeof services[key] === "string" && services[key].includes("грн")) {
            prices.push(parseInt(services[key])); // Зберігаємо ціни в масив
        }
    }
    return Math.max(...prices) + " грн"; // Знаходимо максимальну ціну
}
};
services["Розбити скло"] = "200 грн";


var totalPrice = methods.price();
var minimumPrice = methods.minPrice();
var maximumPrice = methods.maxPrice();


console.log("Total price:", totalPrice);
console.log("Minimum price:", minimumPrice);
console.log("Maximum price:", maximumPrice);
